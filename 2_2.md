Node kth2last(Node head, int k) {
    // 1 2 3 4 5 6 7
    // 3rd to last elem is 5

    // Q: p = q = head
    // R: p = tail.next(null) && len(p-q) = k
    Node p = head;
    Node q = head;

    for (int cnt = 0; cnt < k; cnt++) {
        if (p == null)
            return null;
        else
            p = p.next;
    }

    // p == null || p != null
    while (p != null) {
        q = q.next;
        p = p.next;
    }

    return q;
}
