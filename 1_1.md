    if (str == null || str.length() == 0);

    if (str.length() > 128)
        return false;

    boolean char_set[] = new boolean[128];
    for (char c: str.toCharArray()) {
        int i = c;
        if (char_set[i])
            return false;
        else
            char_set[i] = true;;
    }

    return true;
}
