Node rmDup(Node head) {
    // tmp buffer not allowed
    // Time: len^2

    if (head == null)
        return head;

    Node p = head;
    Node q, prev_q;

    // Q: p != null
    // R: for each node p in list, each following node q != p
    while (p != null) {
        prev_q = p;
        q = p.next;
        while (q != null) {
            if (q.d_char == p.d_char) {
                prev_q.next = q.next;
            } else {
                prev_q = prev_q.next;
            }
            q = q.next;
        }
        p = p.next;
    }

    return head;
}
