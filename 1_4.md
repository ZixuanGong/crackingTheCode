String replaceWS (String s, int len) {

    char arr[] = s.toCharArray();

    int p = len-1;    // "true length"
    for (int i = arr.length-1; i >= 0; i--) {
        char c = s.charAt(p);
        if (c != ' ') {
            arr[i] = c;
        } else {
            arr[i] = '0';
            arr[--i] = '2';
            arr[--i] = '%';
        }
        p--;
    }
    return new String(arr);
}
